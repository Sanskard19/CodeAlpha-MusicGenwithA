03
Preprocess

Preprocessing with music21

Convert raw MIDI into token sequences the model can learn from. Build a vocabulary of notes and chords.

Pipeline: preprocess.py
Parse each MIDI with music21.converter.parse()
Extract note.Note and chord.Chord elements
Encode to string tokens, e.g. "C4", "E4.G4.B4"
Build token2idx / idx2token vocabulary dicts
Slice into (input, target) sequence pairs of length 50–100
Save to data/processed/sequences.npy and vocab.pkl 
# preprocess.py (key excerpt)
from music21 import converter, instrument, note, chord
import pickle, numpy as np

def parse_midi(filepath):
    s = converter.parse(filepath)
    part = instrument.partitionByInstrument(s)
    notes_out = []
    for element in (part.parts[0] if part else s).flat.notes:
        if isinstance(element, note.Note):
            notes_out.append(str(element.pitch))
        elif isinstance(element, chord.Chord):
            notes_out.append(".".join(str(n) for n in element.normalOrder))
    return notes_out

04
Model

Model Design (LSTM)

Build a character-level LSTM that learns to predict the next musical token. Start simple, tune from there.

Architecture
Embedding Layer
vocab_size → embed_dim (128)
LSTM Layer 1
hidden_size=256, dropout=0.3
LSTM Layer 2
hidden_size=256
Dense + Softmax
→ vocab_size
# model.py (PyTorch)
import torch.nn as nn

class MusicLSTM(nn.Module):
    def __init__(self, vocab_size, embed_dim=128,
                 hidden=256, layers=2, dropout=0.3):
        super().__init__()
        self.embed = nn.Embedding(vocab_size, embed_dim)
        self.lstm  = nn.LSTM(embed_dim, hidden, layers,
                              batch_first=True, dropout=dropout)
        self.fc    = nn.Linear(hidden, vocab_size)

    def forward(self, x, hidden=None):
        x = self.embed(x)
        x, hidden = self.lstm(x, hidden)
        return self.fc(x), hidden

Generate

Music Generation Pipeline

Load the trained model, produce new token sequences, and convert them back to playable MIDI files.

# generate.py
import torch
from music21 import stream, note, chord, instrument

def generate(model, seed_seq, vocab, n=200, temperature=1.0):
    model.eval()
    generated = list(seed_seq)
    hidden = None
    with torch.no_grad():
        for _ in range(n):
            x = torch.tensor([[vocab.token2idx[generated[-1]]]])
            out, hidden = model(x, hidden)
            probs = (out[0,-1] / temperature).softmax(-1)
            idx = torch.multinomial(probs, 1).item()
            generated.append(vocab.idx2token[idx])
    return generated

def tokens_to_midi(tokens, output_path="outputs/generated.mid"):
    s = stream.Stream()
    s.append(instrument.Piano())
    offset = 0
    for tok in tokens:
        if "." in tok:          # chord
            ns = [note.Note(int(n)) for n in tok.split(".")]
            c = chord.Chord(ns)
            c.offset = offset; s.append(c)
        else:                    # single note
            n_ = note.Note(tok); n_.offset = offset; s.append(n_)
        offset += 0.5
    s.write("midi", fp=output_path)
